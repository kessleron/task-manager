    <script>

        // Tab switching functionality

        document.addEventListener('DOMContentLoaded', function() {

            const tabButtons = document.querySelectorAll('.tab-button');


            tabButtons.forEach(button => {

                button.addEventListener('click', function() {

                    // Remove active class from all buttons and contents

                    tabButtons.forEach(btn => {

                        btn.classList.remove('active', 'bg-green-500', 'text-white');

                        btn.classList.add('text-gray-700');

                    });


                    document.querySelectorAll('.tabcontent').forEach(content => {

                        content.classList.remove('active');

                    });


                    // Add active class to clicked button and its content

                    this.classList.add('active', 'bg-green-500', 'text-white');

                    this.classList.remove('text-gray-700');


                    const targetId = this.getAttribute('data-target');

                    document.getElementById(targetId).classList.add('active');

                });

            });


            // Initialize first tab as active

            document.querySelector('.tab-button.active').classList.add('bg-green-500', 'text-white');

            document.querySelector('.tab-button.active').classList.remove('text-gray-700');


            // Make file upload area clickable

            const dropZone = document.getElementById('drop-zone');

            const fileInput = document.getElementById('file-input-element');


            if (dropZone && fileInput) {

                dropZone.addEventListener('click', () => {

                    fileInput.click();

                });

            }

        });


        // Update file name when file is selected

        function updateFileName() {

            const fileInput = document.getElementById('file-input-element');

            const fileNameDisplay = document.getElementById('file-name-display');


            if (fileInput && fileInput.files.length > 0) {

                fileNameDisplay.textContent = fileInput.files[0].name;

                fileNameDisplay.classList.add('font-semibold', 'text-green-600');

            } else {

                fileNameDisplay.textContent = 'גרור לכאן קובץ Word או לחץ לבחירת קובץ';

                fileNameDisplay.classList.remove('font-semibold', 'text-green-600');

            }

        }


        // Function to generate HTML for potential tasks (will be used by Flask)

        function renderPotentialTasks(tasks) {

            const container = document.getElementById('potential-tasks-container');

            const section = document.getElementById('potential-tasks-section');

            const countElement = document.getElementById('potential-task-count');


            if (tasks && tasks.length > 0) {

                section.classList.remove('hidden');

                countElement.textContent = tasks.length;


                let html = '';

                tasks.forEach(task => {

                    html += `

                    <div class="potential-task bg-green-50 p-4 mb-4 rounded-md">

                        <h3 class="font-semibold">${task.description}</h3>

                        ${task.due_date ? `<p class="mb-2">תאריך יעד משוער: ${task.due_date}</p>` : ''}

                        <div class="text-gray-500 text-sm mt-2 pt-2 border-t border-gray-200">מקור: ${task.raw_text}</div>

                        <form method="post" class="mt-3">

                            <input type="hidden" name="confirm_task_id" value="${task.id}">

                            <button type="submit" class="bg-green-500 hover:bg-green-600 text-white py-1 px-3 text-sm rounded-md transition-colors">אשר משימה</button>

                        </form>

                    </div>

                    `;

                });


                container.innerHTML = html;

            } else {

                section.classList.add('hidden');

            }

        }


        // Function to generate HTML for tasks (will be used by Flask)

        function renderTasks(tasks) {

            const container = document.getElementById('tasks-container');

            const noTasksMessage = document.getElementById('no-tasks-message');

            const countElement = document.getElementById('task-count');

            const countNumberElement = document.getElementById('tasks-count-number');


            if (tasks && tasks.length > 0) {

                noTasksMessage.classList.add('hidden');

                countElement.classList.remove('hidden');

                countNumberElement.textContent = tasks.length;


                let html = '';

                tasks.forEach(task => {

                    html += `

                    <div class="task-item flex justify-between items-center p-4 border-b border-gray-200">

                        <div class="task-description ${task.done ? 'line-through text-gray-500' : ''} flex-grow">

                            ${task.description}

                            ${task.due_date ? `<span class="mr-2 text-sm text-gray-600">(עד: ${task.due_date})</span>` : ''}

                        </div>

                        <div class="task-actions flex space-x-2">

                            <a href="/toggle/${task.id}" class="bg-blue-500 hover:bg-blue-600 text-white py-1 px-3 text-sm rounded-md transition-colors">

                                ${task.done ? 'בטל סימון' : 'סמן כבוצע'}

                            </a>

                            <a href="/delete/${task.id}" class="bg-red-500 hover:bg-red-600 text-white py-1 px-3 text-sm rounded-md transition-colors">

                                מחק

                            </a>

                        </div>

                    </div>

                    `;

                });


                container.innerHTML = html;

            } else {

                noTasksMessage.classList.remove('hidden');

                countElement.classList.add('hidden');

                container.innerHTML = '<p id="no-tasks-message" class="text-center text-gray-500 py-8">אין משימות כרגע.</p>';

            }

        }
